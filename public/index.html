function processSrt(content, exceptions) { // Your SRT processing logic here const exceptionsSet = new Set(exceptions.map(exception => exception.toLowerCase())); const punctuationRegex = /[^\w\s'"?:]/g; return content .split("\n") .map((line) => { if (line.match(/^\d+$/) || line.includes("-->")) { return line; } return line .replace(/[^\w\s'"?:]/g, (char) => { return char === "'" || char === "-" || char === "?" || char === ":" || char === '"' ? char : ""; }) .replace(/\b\w+\b/g, (word) => { const cleanWord = word.toLowerCase(); if (exceptionsSet.has(cleanWord)) { return word; } return cleanWord; }); }) .join("\n"); }
